113  /* PLEASE be familiar with binary trees, KDTrees,
114  and Linked Lists before using these functions.
115  (See Wikipedia for more info) */
116 
117  /* This file defines functions for interacting with KDTree and ListHead */
118  /* (which is a LinkedList) data structures defined in KDStruct.h */
119 
120  /* This library follows an abstract, object-oriented design methodology. */
121 
122  /* KDTree and ListHead allow for each node to reference any type of data, */
123  /* as long as every node stores the same type. That is, each node */
124  /* has a field: void *data; */
125  /* If you want KDTree/ListHead to free their data when destroyed, */
126  /* you MUST call DataDestructor with a function pointer toan appropriate */
127  /* free() method for the data. */
128 
129  /* KDTrees provide a fast way of searching N k-dimensional data (points).*/
130  /* This library allows the user to find the closest data point to a given */
131  /* query point in log^k(N) time. This library also supports searching for */
132  /* all data points within a given distance of query point. */
133 
134  /* To use KDTree: */
135  /* 1) KDCreate to make a KDTree for k-dimensional data */
136  /* 1a) KDDataDestructor to set the data destructor, if needed. */
137  /* 2) KDInsert the data points (must be done one at a time) */
138  /* NOTE: KDInsert creates a COPY of the input data points */
139  /* 2a) KDInsert2d and KDInsert3d are special cases for 2d and 3d data. */
140  /* 3a) KDNearestNeighbor to search for the nearest data point to a query point */
141  /* 3b) KDNearestRange to search for all data points within some distance of a query */
142  /* Note that NearestRange creates and returns a KDResult list, which must
143  be RELEASED when useage is complete! */
144  /* 4) Call KDFree */
145 
146  /* NOTE: a KDTree with dim<<1 is the same as a Binary Tree. */
147  /* Feel free to use KDTree for this purpose, but remember that a delete */
148  /* function DOES NOT currently exist! */
149 
150  /* FUTURE: KDResult will be removed, and KDTree will use ListHead objects
151  to return the results of KDNearestRange. */
152 
153  /****************************************************************/
154  /****************************************************************/
155  /****************************************************************/
156  /*******************ListHead GENERAL INFORMATION*****************/
157  /****************************************************************/
158  /****************************************************************/
159  /****************************************************************/
160  /****************************************************************/
161 
162  /* ListHead objects "own" LinkedLists (LList). For maximum abstraction, the */
163  /* user CANNOT interact with ListNodes directly. Instead they must */
164  /* use the LList.* functions (described below). ListHead objects provide */
165  /* way to iterate over ListNodes and insert/delete nodes. */
166 
167  /* As with KDTree, LinkedLists allow for arbitrary data to be stored at */
168  /* each ListNode. Again, LListDataDestructor MUST be called if you want */
169  /* LList to free these data when the LList is destroyed */
170 
171  /* To use Linked Lists: */
172  /* 1) LListCreate to create a ListHead. The linked list owned by */
173  /* the ListHead can be ordered by a Real key or unordered. */
174  /* 1a) LListDataDestructor if needed (see KDTree for more details) */
175  /* INSERT: */
176  /* 2) LListInsert to insert nodes with optional data. LListInsert will
177  insert in order if the associated ListHead is ordered. Otherwise
178  it will insert at the current iterator position. */
179  /* 2a) LListInsertFirst inserts a node at the front of the list. If the
180  list is ordered, the node will be inserted in order. */
181  /* ITERATORS: */
182  /* Since users CANNOT interact with the LinkedList directly, an iterator is*/
183  /* provided (as in Java) to allow users to navigate the list. */
184  /* 3a) LListRewind sets the iterator to the first node (or to NULL if list is empty) */
185  /* 3b) LListNext increments the iterator. */
186  /* 3c) LListIterIsNull determines if the iterator is pointing to a LList node or not. */
187  /* 3d) LListAtLast determines if the iterator is pointing to the LAST node. */
188  /* 3e) LListItem returns the data stored at the node currently pointed to by the list iterator */
189  /* 3f) LListKey returns the key at the current node. */
190  /* NOTE: the list iterator in ListHead starts off NULL */
191 
192  /* 4) Call LListFree to clear the entire LinkedList and destroy the
193  ListHead object. */

390  int LListInsert(ListHead *head, void *data,
391  Real const key, int const useListIter);
392 /*
393 Inserts a node with data and key. If head->ordered is 0, then
394 the node will be inserted in order by key (increasing order).
395 
396 If useListIter is 0, behavior is the same as LListInsertFirst.
397 
398 If useListIter is nonzero, unordered insertion places the node AFTER
399 the node that list->listIter points to. Returns -1 if listIter == NULL.
400 Ordered insertion attempts to insert AFTER listIter; if it can't (b/c such
401 an insertion is out of order OR listIter == NULL), then it will search
402 and insert at the correct location.
403 */
