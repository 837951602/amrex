464  team_range (int begin, int end, int rit = -1, int nworkers = 0)
465  {
466  int rb, re;
467  {
468  if (rit < 0) rit = ParallelDescriptor::MyRankInTeam();
469  if (nworkers == 0) nworkers = ParallelDescriptor::TeamSize();
470  BL_ASSERT(rit<nworkers);
471  if (nworkers == 1) {
472  rb = begin;
473  re = end;
474  } else {
475  int ntot = end - begin;
476  int nr = ntot / nworkers;
477  int nlft = ntot - nr * nworkers;
478  if (rit < nlft) { // get nr+1 items
479  rb = begin + rit * (nr + 1);
480  re = rb + nr + 1;
481  } else { // get nr items
482  rb = begin + rit * nr + nlft;
483  re = rb + nr;
484  }

492  int ntot = re - rb;
493  int nr = ntot / nthreads;
494  int nlft = ntot - nr * nthreads;
495  if (tid < nlft) { // get nr+1 items
496  rb += tid * (nr + 1);
497  re = rb + nr + 1;
498  } else { // get nr items
499  rb += tid * nr + nlft;
500  re = rb + nr;
501  }
