648  Real BrentRootFinder(const RvDim & a_loPt,
649  const RvDim & a_hiPt,
650  const int & a_edgeDir) const
651  {
652  // Max allowed iterations and floating point precision
653  const unsigned int MAXITER = 100;
654  const Real EPS = 3.0e-15;
655 
656  unsigned int i;
657  Real aPt;
658  Real bPt;
659  Real c, fa, fb, fc;
660  Real d, e;
661  Real tol1, xm;
662  Real p, q, r, s;
663 
664  aPt = -1.0; // a_smallestRoot;
665  bPt = 1.0; // a_biggestRoot;
666 
667  RvDim physCoordAPt = a_loPt;
668  RvDim physCoordBPt = a_hiPt;
669 
670  fa = -m_function->value(IndexTM<int,dim>::Zero,physCoordAPt);
671  fb = -m_function->value(IndexTM<int,dim>::Zero,physCoordBPt);
672 
673  // Init these to be safe
674  c = d = e = 0.0;
675 
676  if (fb*fa > 0)
677  {
678  amrex::pout() << "fa " << fa << " fb " << fb <<endl;
679  amrex::Abort("IFData::BrentRootFinder. Root must be bracketed, but instead the supplied end points have the same sign.");
680  }
681 
682  fc = fb;
683 
684  for (i = 0; i < MAXITER; i++)
685  {
686  if (fb*fc > 0)
687  {
688  // Rename a, b, c and adjust bounding interval d
689  c = aPt;
690  fc = fa;
691  d = bPt - aPt;
692  e = d;
693  }
694 
695  if (std::abs(fc) < std::abs(fb))
696  {
697  aPt = bPt;
698  bPt = c;
699  c = aPt;
700  fa = fb;
701  fb = fc;
702  fc = fa;
703  }
704 
705  // Convergence check
706  tol1 = 2.0 * EPS * std::abs(bPt) + 0.5 * TOLERANCE;
707  xm = 0.5 * (c - bPt);
708 
709  if (std::abs(xm) <= tol1 || fb == 0.0)
710  {
711  break;
712  }
713 
714  if (std::abs(e) >= tol1 && std::abs(fa) > std::abs(fb))
715  {
716  // Attempt inverse quadratic interpolation
717  s = fb / fa;
718  if (aPt == c)
719  {
720  p = 2.0 * xm * s;
721  q = 1.0 - s;
722  }
723  else
724  {
725  q = fa / fc;
726  r = fb / fc;
727  p = s * (2.0 * xm * q * (q-r) - (bPt-aPt) * (r-1.0));
728  q = (q-1.0) * (r-1.0) * (s-1.0);
729  }
730 
731  // Check whether in bounds
732  if (p > 0) q = -q;
733 
734  p = std::abs(p);
735 
736  if (2.0 * p < std::min(((Real)3.0)*xm*q-std::abs(tol1*q), std::abs(e*q)))
737  {
738  // Accept interpolation
739  e = d;
740  d = p / q;
741  }
742  else
743  {
744  // Interpolation failed, use bisection
745  d = xm;
746  e = d;
747  }
748  }
749  else

752  d = xm;
753  e = d;
754  }
755

758  fa = fb;
759

764  }

768  else bPt = bPt + tol1;
769  }

773  }
774 
775  if (i >= MAXITER)
776  {
777  cerr << "BrentRootFinder: exceeding maximum iterations: "
778  << MAXITER
779  << "\n";
780  }
781 
782  return bPt;
783  }
